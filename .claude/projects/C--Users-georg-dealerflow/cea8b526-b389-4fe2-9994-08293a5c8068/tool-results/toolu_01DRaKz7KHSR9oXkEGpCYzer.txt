     1→/**
     2→ * BottomSheet Component - Mobile-friendly modal that slides up from bottom
     3→ * Usage:
     4→ *   <BottomSheet isOpen={isOpen} onClose={() => setIsOpen(false)} title="Filters">
     5→ *     <FilterContent />
     6→ *   </BottomSheet>
     7→ */
     8→
     9→import { useEffect, useRef } from "react";
    10→
    11→export function BottomSheet({
    12→  isOpen,
    13→  onClose,
    14→  title,
    15→  children,
    16→  footer,
    17→  maxHeight = "85vh",
    18→  className = "",
    19→}) {
    20→  const sheetRef = useRef(null);
    21→
    22→  // Lock body scroll when open
    23→  useEffect(() => {
    24→    if (isOpen) {
    25→      document.body.style.overflow = "hidden";
    26→    } else {
    27→      document.body.style.overflow = "";
    28→    }
    29→    return () => {
    30→      document.body.style.overflow = "";
    31→    };
    32→  }, [isOpen]);
    33→
    34→  // Handle escape key
    35→  useEffect(() => {
    36→    const handleEscape = (e) => {
    37→      if (e.key === "Escape" && isOpen) {
    38→        onClose();
    39→      }
    40→    };
    41→    window.addEventListener("keydown", handleEscape);
    42→    return () => window.removeEventListener("keydown", handleEscape);
    43→  }, [isOpen, onClose]);
    44→
    45→  if (!isOpen) return null;
    46→
    47→  return (
    48→    <div className="fixed inset-0 z-50 md:hidden">
    49→      {/* Backdrop */}
    50→      <div
    51→        className="absolute inset-0 bg-black/50 backdrop-blur-sm"
    52→        onClick={onClose}
    53→      />
    54→
    55→      {/* Sheet */}
    56→      <div
    57→        ref={sheetRef}
    58→        className={`absolute bottom-0 left-0 right-0 bg-white rounded-t-2xl shadow-2xl flex flex-col animate-slide-up ${className}`}
    59→        style={{ maxHeight }}
    60→      >
    61→        {/* Handle */}
    62→        <div className="flex justify-center py-2">
    63→          <div className="w-10 h-1 bg-slate-300 rounded-full" />
    64→        </div>
    65→
    66→        {/* Header */}
    67→        <div className="flex items-center justify-between px-4 pb-3 border-b border-slate-200">
    68→          <h2 className="text-lg font-bold text-slate-900">{title}</h2>
    69→          <button
    70→            onClick={onClose}
    71→            className="p-2 -mr-2 rounded-lg text-slate-400 hover:text-slate-600 hover:bg-slate-100 transition-colors"
    72→          >
    73→            <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
    74→              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
    75→            </svg>
    76→          </button>
    77→        </div>
    78→
    79→        {/* Content - Scrollable */}
    80→        <div className="flex-1 overflow-y-auto overscroll-contain p-4">
    81→          {children}
    82→        </div>
    83→
    84→        {/* Footer - Sticky */}
    85→        {footer && (
    86→          <div className="sticky bottom-0 border-t border-slate-200 bg-white p-4 safe-area-bottom">
    87→            {footer}
    88→          </div>
    89→        )}
    90→      </div>
    91→
    92→      <style jsx global>{`
    93→        @keyframes slide-up {
    94→          from {
    95→            transform: translateY(100%);
    96→          }
    97→          to {
    98→            transform: translateY(0);
    99→          }
   100→        }
   101→        .animate-slide-up {
   102→          animation: slide-up 0.3s ease-out forwards;
   103→        }
   104→        .safe-area-bottom {
   105→          padding-bottom: max(1rem, env(safe-area-inset-bottom));
   106→        }
   107→      `}</style>
   108→    </div>
   109→  );
   110→}
   111→
   112→export default BottomSheet;
