     1→/**
     2→ * Badge Component - Status pills and chips
     3→ * Usage:
     4→ *   <Badge variant="success">Active</Badge>
     5→ *   <Badge variant="warning" size="lg">Pending</Badge>
     6→ */
     7→
     8→const VARIANTS = {
     9→  // Neutral
    10→  default: "bg-slate-100 text-slate-700 border-slate-200",
    11→  secondary: "bg-slate-50 text-slate-600 border-slate-200",
    12→
    13→  // Semantic
    14→  success: "bg-emerald-100 text-emerald-700 border-emerald-200",
    15→  warning: "bg-amber-100 text-amber-700 border-amber-200",
    16→  danger: "bg-red-100 text-red-700 border-red-200",
    17→  info: "bg-blue-100 text-blue-700 border-blue-200",
    18→
    19→  // Brand
    20→  primary: "bg-violet-100 text-violet-700 border-violet-200",
    21→  accent: "bg-cyan-100 text-cyan-700 border-cyan-200",
    22→
    23→  // Special
    24→  outline: "bg-transparent text-slate-600 border-slate-300",
    25→  ghost: "bg-transparent text-slate-500 border-transparent",
    26→};
    27→
    28→const SIZES = {
    29→  xs: "px-1.5 py-0.5 text-[10px]",
    30→  sm: "px-2 py-0.5 text-xs",
    31→  default: "px-2.5 py-1 text-xs",
    32→  lg: "px-3 py-1.5 text-sm",
    33→};
    34→
    35→export function Badge({
    36→  children,
    37→  variant = "default",
    38→  size = "default",
    39→  dot = false,
    40→  dotColor,
    41→  icon,
    42→  className = "",
    43→  ...props
    44→}) {
    45→  const dotColors = {
    46→    success: "bg-emerald-500",
    47→    warning: "bg-amber-500",
    48→    danger: "bg-red-500",
    49→    info: "bg-blue-500",
    50→    primary: "bg-violet-500",
    51→    default: "bg-slate-500",
    52→  };
    53→
    54→  return (
    55→    <span
    56→      className={`inline-flex items-center gap-1 rounded-full font-semibold border capitalize ${VARIANTS[variant] || VARIANTS.default} ${SIZES[size] || SIZES.default} ${className}`}
    57→      {...props}
    58→    >
    59→      {dot && (
    60→        <span className={`w-1.5 h-1.5 rounded-full ${dotColors[dotColor || variant] || dotColors.default}`} />
    61→      )}
    62→      {icon && <span className="flex-shrink-0">{icon}</span>}
    63→      {children}
    64→    </span>
    65→  );
    66→}
    67→
    68→/**
    69→ * StatusBadge - Pre-configured badges for common statuses
    70→ */
    71→export function StatusBadge({ status, size = "default" }) {
    72→  const statusConfig = {
    73→    // General
    74→    active: { variant: "success", label: "Active" },
    75→    inactive: { variant: "secondary", label: "Inactive" },
    76→    pending: { variant: "warning", label: "Pending" },
    77→    completed: { variant: "success", label: "Completed" },
    78→    cancelled: { variant: "danger", label: "Cancelled" },
    79→
    80→    // Vehicle
    81→    in_stock: { variant: "success", label: "In Stock" },
    82→    in_prep: { variant: "warning", label: "In Prep" },
    83→    live: { variant: "info", label: "Live" },
    84→    sold: { variant: "primary", label: "Sold" },
    85→    delivered: { variant: "success", label: "Delivered" },
    86→
    87→    // Warranty
    88→    not_booked_in: { variant: "warning", label: "Not Booked" },
    89→    booked_in: { variant: "info", label: "Booked In" },
    90→    on_site: { variant: "primary", label: "On Site" },
    91→    work_complete: { variant: "success", label: "Complete" },
    92→    collected: { variant: "success", label: "Collected" },
    93→
    94→    // Priority
    95→    low: { variant: "secondary", label: "Low" },
    96→    normal: { variant: "default", label: "Normal" },
    97→    high: { variant: "warning", label: "High" },
    98→    critical: { variant: "danger", label: "Critical" },
    99→
   100→    // Issues
   101→    outstanding: { variant: "danger", label: "Outstanding" },
   102→    ordered: { variant: "warning", label: "Ordered" },
   103→    in_progress: { variant: "info", label: "In Progress" },
   104→    resolved: { variant: "success", label: "Resolved" },
   105→  };
   106→
   107→  const normalizedStatus = status?.toLowerCase().replace(/\s+/g, "_");
   108→  const config = statusConfig[normalizedStatus] || { variant: "default", label: status };
   109→
   110→  return (
   111→    <Badge variant={config.variant} size={size}>
   112→      {config.label}
   113→    </Badge>
   114→  );
   115→}
   116→
   117→export default Badge;
