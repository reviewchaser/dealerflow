    29→// Sold statuses - these show "Sold X days" instead of "In stock X days"
    30→const SOLD_STATUSES = ["live", "reserved", "delivered"];
    31→
    32→// Helper to compute duration label based on status
    33→const getVehicleDuration = (vehicle) => {
    34→  const isSold = SOLD_STATUSES.includes(vehicle.status);
    35→
    36→  if (isSold) {
    37→    if (vehicle.soldAt) {
    38→      const daysSold = Math.floor((Date.now() - new Date(vehicle.soldAt).getTime()) / (1000 * 60 * 60 * 24));
    39→      return { days: daysSold, label: `Sold ${daysSold}d`, isSold: true };
    40→    }
    41→    // Legacy sold vehicle without soldAt - just show "Sold"
    42→    return { days: 0, label: "Sold", isSold: true };
    43→  }
    44→
    45→  const daysInStock = vehicle.createdAt
    46→    ? Math.floor((Date.now() - new Date(vehicle.createdAt).getTime()) / (1000 * 60 * 60 * 24))
    47→    : 0;
    48→  return { days: daysInStock, label: `${daysInStock}d in stock`, isSold: false };
    49→};
