     1→import { useState, useEffect, useCallback } from "react";
     2→import { toast } from "react-hot-toast";
     3→import ContactPicker from "@/components/ContactPicker";
     4→
     5→// Format currency
     6→const formatCurrency = (amount) => {
     7→  if (amount === null || amount === undefined) return "—";
     8→  return new Intl.NumberFormat("en-GB", {
     9→    style: "currency",
    10→    currency: "GBP",
    11→  }).format(amount);
    12→};
    13→
    14→// Format date helper
    15→const formatDate = (date) => {
    16→  if (!date) return "";
    17→  return new Date(date).toLocaleDateString("en-GB", {
    18→    day: "numeric",
    19→    month: "short",
    20→    year: "numeric",
    21→  });
    22→};
    23→
    24→// Format datetime helper
    25→const formatDateTime = (date) => {
    26→  if (!date) return "";
    27→  return new Date(date).toLocaleString("en-GB", {
    28→    day: "numeric",
    29→    month: "short",
    30→    year: "numeric",
    31→    hour: "2-digit",
    32→    minute: "2-digit",
    33→  });
    34→};
    35→
    36→// Status labels and colors
    37→const STATUS_CONFIG = {
    38→  DRAFT: { label: "Draft", color: "bg-slate-100 text-slate-600" },
    39→  DEPOSIT_TAKEN: { label: "Deposit Taken", color: "bg-amber-100 text-amber-700" },
    40→  INVOICED: { label: "Invoiced", color: "bg-blue-100 text-blue-700" },
    41→  DELIVERED: { label: "Delivered", color: "bg-purple-100 text-purple-700" },
    42→  COMPLETED: { label: "Completed", color: "bg-emerald-100 text-emerald-700" },
    43→  CANCELLED: { label: "Cancelled", color: "bg-red-100 text-red-700" },
    44→};
    45→
    46→const VAT_SCHEME_LABELS = {
    47→  MARGIN: "Margin Scheme",
    48→  VAT_QUALIFYING: "VAT Qualifying",
    49→  NO_VAT: "No VAT",
    50→};
    51→
    52→/**
    53→ * DealDrawer - Display and manage deal details
    54→ *
    55→ * Props:
    56→ * - dealId: The ID of the deal to display
    57→ * - isOpen: Whether the drawer is open
    58→ * - onClose: Function to call when closing the drawer
    59→ * - onUpdate: Function to call when deal is updated
    60→ */
    61→export default function DealDrawer({
    62→  dealId,
    63→  isOpen,
    64→  onClose,
    65→  onUpdate,
    66→}) {
    67→  const [deal, setDeal] = useState(null);
    68→  const [isLoading, setIsLoading] = useState(false);
    69→  const [activeTab, setActiveTab] = useState("overview");
    70→  const [actionLoading, setActionLoading] = useState(null);
    71→
    72→  // Deposit modal state
    73→  const [showDepositModal, setShowDepositModal] = useState(false);
    74→  const [depositAmount, setDepositAmount] = useState("");
    75→  const [depositMethod, setDepositMethod] = useState("CARD");
    76→  const [depositReference, setDepositReference] = useState("");
    77→
    78→  // Customer editing state
    79→  const [isEditingCustomer, setIsEditingCustomer] = useState(false);
    80→
    81→  // Lock body scroll when drawer is open
    82→  useEffect(() => {
    83→    if (isOpen) {
    84→      const scrollY = window.scrollY;
    85→      const html = document.documentElement;
    86→
    87→      document.body.style.position = "fixed";
    88→      document.body.style.top = `-${scrollY}px`;
    89→      document.body.style.left = "0";
    90→      document.body.style.right = "0";
    91→      document.body.style.width = "100%";
    92→      document.body.style.overflow = "hidden";
    93→      document.body.style.touchAction = "none";
    94→
    95→      html.style.overflow = "hidden";
    96→
    97→      return () => {
    98→        document.body.style.position = "";
    99→        document.body.style.top = "";
   100→        document.body.style.left = "";
   101→        document.body.style.right = "";
   102→        document.body.style.width = "";
   103→        document.body.style.overflow = "";
   104→        document.body.style.touchAction = "";
   105→
   106→        html.style.overflow = "";
   107→
   108→        window.scrollTo(0, scrollY);
   109→      };
   110→    }
   111→  }, [isOpen]);
   112→
   113→  useEffect(() => {
   114→    if (isOpen && dealId) {
   115→      fetchDeal();
   116→    }
   117→  }, [isOpen, dealId]);
   118→
   119→  const fetchDeal = async () => {
   120→    setIsLoading(true);
   121→    try {
   122→      const res = await fetch(`/api/deals/${dealId}`);
   123→      if (!res.ok) throw new Error("Failed to load deal");
   124→      const data = await res.json();
   125→      setDeal(data);
   126→    } catch (error) {
   127→      toast.error("Failed to load deal details");
   128→      console.error(error);
   129→    } finally {
   130→      setIsLoading(false);
   131→    }
   132→  };
   133→
   134→  // Calculate totals
   135→  const calculateTotals = () => {
   136→    if (!deal) return {};
   137→
   138→    const addOnsNetTotal = (deal.addOns || []).reduce(
   139→      (sum, a) => sum + (a.unitPriceNet * (a.qty || 1)),
   140→      0
   141→    );
   142→    const addOnsVatTotal = (deal.addOns || []).reduce((sum, a) => {
   143→      if (a.vatTreatment === "STANDARD") {
   144→        return sum + a.unitPriceNet * (a.qty || 1) * (a.vatRate || 0.2);
   145→      }
   146→      return sum;
   147→    }, 0);
   148→
   149→    let subtotal, totalVat, grandTotal;
   150→
   151→    if (deal.vatScheme === "VAT_QUALIFYING") {
   152→      subtotal = (deal.vehiclePriceNet || 0) + addOnsNetTotal;
   153→      totalVat = (deal.vehicleVatAmount || 0) + addOnsVatTotal;
   154→      grandTotal = subtotal + totalVat;
   155→    } else {
   156→      subtotal = (deal.vehiclePriceGross || 0) + addOnsNetTotal + addOnsVatTotal;
   157→      totalVat = 0;
   158→      grandTotal = subtotal;
   159→    }
   160→
   161→    const totalPaid = (deal.payments || [])
   162→      .filter((p) => !p.isRefunded)
   163→      .reduce((sum, p) => sum + p.amount, 0);
   164→
   165→    const pxNetValue = deal.partExchange
   166→      ? (deal.partExchange.allowance || 0) - (deal.partExchange.settlement || 0)
   167→      : 0;
   168→
   169→    const balanceDue = grandTotal - totalPaid - pxNetValue;
   170→
   171→    return {
   172→      addOnsNetTotal,
   173→      addOnsVatTotal,
   174→      subtotal,
   175→      totalVat,
   176→      grandTotal,
   177→      totalPaid,
   178→      pxNetValue,
   179→      balanceDue,
   180→    };
   181→  };
   182→
   183→  // Deal actions
   184→  const handleTakeDeposit = async () => {
   185→    if (!depositAmount || parseFloat(depositAmount) <= 0) {
   186→      toast.error("Please enter a valid deposit amount");
   187→      return;
   188→    }
   189→
   190→    setActionLoading("deposit");
   191→    try {
   192→      const res = await fetch(`/api/deals/${dealId}/take-deposit`, {
   193→        method: "POST",
   194→        headers: { "Content-Type": "application/json" },
   195→        body: JSON.stringify({
   196→          amount: parseFloat(depositAmount),
   197→          method: depositMethod,
   198→          reference: depositReference,
   199→        }),
   200→      });
   201→
   202→      if (!res.ok) {
   203→        const err = await res.json();
   204→        throw new Error(err.error || "Failed to record deposit");
   205→      }
   206→
   207→      const result = await res.json();
   208→      toast.success(`Deposit recorded: ${result.documentNumber}`);
   209→      setShowDepositModal(false);
   210→      setDepositAmount("");
   211→      setDepositReference("");
   212→      fetchDeal();
   213→      onUpdate?.();
   214→    } catch (error) {
   215→      toast.error(error.message);
   216→    } finally {
   217→      setActionLoading(null);
   218→    }
   219→  };
   220→
   221→  const handleGenerateInvoice = async () => {
   222→    setActionLoading("invoice");
   223→    try {
   224→      const res = await fetch(`/api/deals/${dealId}/generate-invoice`, {
   225→        method: "POST",
   226→        headers: { "Content-Type": "application/json" },
   227→      });
   228→
   229→      if (!res.ok) {
   230→        const err = await res.json();
   231→        throw new Error(err.error || "Failed to generate invoice");
   232→      }
   233→
   234→      const result = await res.json();
   235→      toast.success(`Invoice generated: ${result.documentNumber}`);
   236→      fetchDeal();
   237→      onUpdate?.();
   238→    } catch (error) {
   239→      toast.error(error.message);
   240→    } finally {
   241→      setActionLoading(null);
   242→    }
   243→  };
   244→
   245→  const handleMarkDelivered = async () => {
   246→    setActionLoading("delivered");
   247→    try {
   248→      const res = await fetch(`/api/deals/${dealId}/mark-delivered`, {
   249→        method: "POST",
   250→        headers: { "Content-Type": "application/json" },
   251→      });
   252→
   253→      if (!res.ok) {
   254→        const err = await res.json();
   255→        throw new Error(err.error || "Failed to mark as delivered");
   256→      }
   257→
   258→      toast.success("Deal marked as delivered");
   259→      fetchDeal();
   260→      onUpdate?.();
   261→    } catch (error) {
   262→      toast.error(error.message);
   263→    } finally {
   264→      setActionLoading(null);
   265→    }
   266→  };
   267→
   268→  const handleMarkCompleted = async () => {
   269→    setActionLoading("completed");
   270→    try {
   271→      const res = await fetch(`/api/deals/${dealId}/mark-completed`, {
   272→        method: "POST",
   273→        headers: { "Content-Type": "application/json" },
   274→      });
   275→
   276→      if (!res.ok) {
   277→        const err = await res.json();
   278→        throw new Error(err.error || "Failed to complete deal");
   279→      }
   280→
   281→      toast.success("Deal completed");
   282→      fetchDeal();
   283→      onUpdate?.();
   284→    } catch (error) {
   285→      toast.error(error.message);
   286→    } finally {
   287→      setActionLoading(null);
   288→    }
   289→  };
   290→
   291→  const handleCancelDeal = async () => {
   292→    if (!confirm("Are you sure you want to cancel this deal?")) return;
   293→
   294→    setActionLoading("cancel");
   295→    try {
   296→      const res = await fetch(`/api/deals/${dealId}`, {
   297→        method: "DELETE",
   298→        headers: { "Content-Type": "application/json" },
   299→        body: JSON.stringify({ cancelReason: "Cancelled by user" }),
   300→      });
   301→
   302→      if (!res.ok) {
   303→        const err = await res.json();
   304→        throw new Error(err.error || "Failed to cancel deal");
   305→      }
   306→
   307→      toast.success("Deal cancelled");
   308→      fetchDeal();
   309→      onUpdate?.();
   310→    } catch (error) {
   311→      toast.error(error.message);
   312→    } finally {
   313→      setActionLoading(null);
   314→    }
   315→  };
   316→
   317→  const handleUpdateCustomer = async (contactId) => {
   318→    if (!contactId) return;
   319→
   320→    setActionLoading("customer");
   321→    try {
   322→      const res = await fetch(`/api/deals/${dealId}`, {
   323→        method: "PUT",
   324→        headers: { "Content-Type": "application/json" },
   325→        body: JSON.stringify({ soldToContactId: contactId }),
   326→      });
   327→
   328→      if (!res.ok) {
   329→        const err = await res.json();
   330→        throw new Error(err.error || "Failed to update customer");
   331→      }
   332→
   333→      toast.success("Customer updated");
   334→      setIsEditingCustomer(false);
   335→      fetchDeal();
   336→      onUpdate?.();
   337→    } catch (error) {
   338→      toast.error(error.message);
   339→    } finally {
   340→      setActionLoading(null);
   341→    }
   342→  };
   343→
   344→  if (!isOpen) return null;
   345→
   346→  const totals = calculateTotals();
   347→  const statusConfig = STATUS_CONFIG[deal?.status] || STATUS_CONFIG.DRAFT;
   348→
   349→  return (
   350→    <div
   351→      className="fixed inset-0 flex justify-end z-50 drawer-locked"
   352→      style={{ touchAction: "none", overscrollBehavior: "none" }}
   353→    >
   354→      {/* Backdrop */}
   355→      <div
   356→        className="absolute inset-0 bg-black/40 backdrop-blur-sm transition-opacity duration-300"
   357→        onClick={onClose}
   358→      />
   359→
   360→      {/* Drawer */}
   361→      <div
   362→        className="relative bg-[#f8fafc] flex flex-col w-full max-w-3xl min-w-0 translate-x-0 overflow-x-hidden shadow-2xl"
   363→        style={{
   364→          height: "100dvh",
   365→          maxHeight: "100dvh",
   366→          touchAction: "pan-y",
   367→        }}
   368→      >
   369→        {isLoading ? (
   370→          <div className="flex items-center justify-center h-full">
   371→            <span className="loading loading-spinner loading-lg"></span>
   372→          </div>
   373→        ) : !deal ? (
   374→          <div className="flex flex-col items-center justify-center h-full">
   375→            <p className="text-base-content/60">Deal not found</p>
   376→            <button className="btn btn-ghost btn-sm mt-2" onClick={onClose}>
   377→              Close
   378→            </button>
   379→          </div>
   380→        ) : (
   381→          <>
   382→            {/* Header */}
   383→            <div className="sticky top-0 bg-white border-b border-slate-100 px-4 md:px-6 py-4 md:py-5 z-10 shrink-0 shadow-sm">
   384→              <div className="flex items-center justify-between">
   385→                <div className="flex items-center gap-3">
   386→                  <button
   387→                    className="flex items-center justify-center w-9 h-9 rounded-xl bg-slate-100 hover:bg-[#0066CC]/10 text-slate-500 hover:text-[#0066CC] transition-all"
   388→                    onClick={onClose}
   389→                  >
   390→                    <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   391→                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M15 19l-7-7 7-7" />
   392→                    </svg>
   393→                  </button>
   394→                  <div>
   395→                    <h2 className="text-lg md:text-xl font-bold text-slate-900">
   396→                      Deal #{deal.dealNumber || "—"}
   397→                    </h2>
   398→                    <div className="flex items-center gap-2 mt-1">
   399→                      <span className={`inline-flex items-center px-2.5 py-1 rounded-lg text-xs font-semibold ${statusConfig.color}`}>
   400→                        {statusConfig.label}
   401→                      </span>
   402→                      {deal.vehicle && (
   403→                        <span className="inline-flex items-center px-2.5 py-1 rounded-lg bg-slate-900 text-white text-xs font-mono font-bold tracking-wider">
   404→                          {deal.vehicle.regCurrent}
   405→                        </span>
   406→                      )}
   407→                    </div>
   408→                  </div>
   409→                </div>
   410→                <button
   411→                  className="flex items-center justify-center w-9 h-9 rounded-xl bg-slate-100 hover:bg-red-50 text-slate-400 hover:text-red-500 transition-all"
   412→                  onClick={onClose}
   413→                >
   414→                  <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   415→                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
   416→                  </svg>
   417→                </button>
   418→              </div>
   419→            </div>
   420→
   421→            {/* Tabs */}
   422→            <div className="sticky top-[73px] md:top-[89px] bg-white border-b border-slate-100 px-4 py-3 z-10 shrink-0">
   423→              {/* Mobile: Dropdown selector */}
   424→              <div className="md:hidden">
   425→                <select
   426→                  value={activeTab}
   427→                  onChange={(e) => setActiveTab(e.target.value)}
   428→                  className="select select-bordered w-full font-medium"
   429→                >
   430→                  <option value="overview">Overview</option>
   431→                  <option value="customer">Customer</option>
   432→                  <option value="addons">Add-ons ({deal.addOns?.length || 0})</option>
   433→                  <option value="payments">Payments ({deal.payments?.length || 0})</option>
   434→                </select>
   435→              </div>
   436→
   437→              {/* Desktop: Pill tabs */}
   438→              <div className="hidden md:flex items-center gap-2 overflow-x-auto pb-1 -mb-1">
   439→                {[
   440→                  { id: "overview", label: "Overview" },
   441→                  { id: "customer", label: "Customer" },
   442→                  { id: "addons", label: "Add-ons", count: deal.addOns?.length },
   443→                  { id: "payments", label: "Payments", count: deal.payments?.length },
   444→                ].map((tab) => (
   445→                  <button
   446→                    key={tab.id}
   447→                    onClick={() => setActiveTab(tab.id)}
   448→                    className={`flex items-center gap-2 px-4 py-2 rounded-xl text-sm font-medium whitespace-nowrap transition-all ${
   449→                      activeTab === tab.id
   450→                        ? "bg-[#0066CC] text-white shadow-md shadow-[#0066CC]/25"
   451→                        : "bg-slate-100 text-slate-600 hover:bg-slate-200 hover:text-slate-900"
   452→                    }`}
   453→                  >
   454→                    <span>{tab.label}</span>
   455→                    {tab.count > 0 && (
   456→                      <span className={`inline-flex items-center justify-center min-w-[20px] h-5 px-1.5 rounded-full text-xs font-bold ${
   457→                        activeTab === tab.id
   458→                          ? "bg-white/20 text-white"
   459→                          : "bg-slate-200 text-slate-600"
   460→                      }`}>
   461→                        {tab.count}
   462→                      </span>
   463→                    )}
   464→                  </button>
   465→                ))}
   466→              </div>
   467→            </div>
   468→
   469→            {/* Content */}
   470→            <div
   471→              className="p-4 md:p-6 space-y-4 md:space-y-6 flex-1 overflow-y-auto overflow-x-hidden min-w-0 overscroll-contain"
   472→              style={{ paddingBottom: "calc(96px + env(safe-area-inset-bottom, 0px))" }}
   473→            >
   474→              {/* Overview Tab */}
   475→              {activeTab === "overview" && (
   476→                <div className="space-y-4">
   477→                  {/* Action Buttons */}
   478→                  {deal.status !== "COMPLETED" && deal.status !== "CANCELLED" && (
   479→                    <div className="bg-white rounded-2xl border border-slate-100 shadow-sm overflow-hidden">
   480→                      <div className="px-4 py-3 bg-gradient-to-r from-slate-50 to-white border-b border-slate-100">
   481→                        <h3 className="text-sm font-bold text-slate-800">Actions</h3>
   482→                      </div>
   483→                      <div className="p-4 flex flex-wrap gap-2">
   484→                        {deal.status === "DRAFT" && (
   485→                          <button
   486→                            onClick={() => setShowDepositModal(true)}
   487→                            disabled={actionLoading}
   488→                            className="btn btn-sm bg-amber-500 hover:bg-amber-600 text-white border-none"
   489→                          >
   490→                            {actionLoading === "deposit" ? (
   491→                              <span className="loading loading-spinner loading-xs"></span>
   492→                            ) : (
   493→                              "Take Deposit"
   494→                            )}
   495→                          </button>
   496→                        )}
   497→                        {(deal.status === "DEPOSIT_TAKEN" || deal.status === "DRAFT") && deal.soldToContactId && deal.vehiclePriceGross && (
   498→                          <button
   499→                            onClick={handleGenerateInvoice}
   500→                            disabled={actionLoading}
   501→                            className="btn btn-sm bg-blue-500 hover:bg-blue-600 text-white border-none"
   502→                          >
   503→                            {actionLoading === "invoice" ? (
   504→                              <span className="loading loading-spinner loading-xs"></span>
   505→                            ) : (
   506→                              "Generate Invoice"
   507→                            )}
   508→                          </button>
   509→                        )}
   510→                        {deal.status === "INVOICED" && (
   511→                          <button
   512→                            onClick={handleMarkDelivered}
   513→                            disabled={actionLoading}
   514→                            className="btn btn-sm bg-purple-500 hover:bg-purple-600 text-white border-none"
   515→                          >
   516→                            {actionLoading === "delivered" ? (
   517→                              <span className="loading loading-spinner loading-xs"></span>
   518→                            ) : (
   519→                              "Mark Delivered"
   520→                            )}
   521→                          </button>
   522→                        )}
   523→                        {deal.status === "DELIVERED" && (
   524→                          <button
   525→                            onClick={handleMarkCompleted}
   526→                            disabled={actionLoading}
   527→                            className="btn btn-sm bg-emerald-500 hover:bg-emerald-600 text-white border-none"
   528→                          >
   529→                            {actionLoading === "completed" ? (
   530→                              <span className="loading loading-spinner loading-xs"></span>
   531→                            ) : (
   532→                              "Complete Deal"
   533→                            )}
   534→                          </button>
   535→                        )}
   536→                        <button
   537→                          onClick={handleCancelDeal}
   538→                          disabled={actionLoading}
   539→                          className="btn btn-sm btn-ghost text-red-500 hover:bg-red-50"
   540→                        >
   541→                          Cancel Deal
   542→                        </button>
   543→                      </div>
   544→                    </div>
   545→                  )}
   546→
   547→                  {/* Vehicle Section */}
   548→                  {deal.vehicle && (
   549→                    <div className="bg-white rounded-2xl border border-slate-100 shadow-sm overflow-hidden">
   550→                      <div className="px-4 py-3 bg-gradient-to-r from-slate-50 to-white border-b border-slate-100">
   551→                        <div className="flex items-center gap-2">
   552→                          <div className="w-8 h-8 rounded-lg bg-slate-100 flex items-center justify-center">
   553→                            <svg className="w-4 h-4 text-slate-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   554→                              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M8 7h8m-8 5h8m-4-9a9 9 0 110 18 9 9 0 010-18z" />
   555→                            </svg>
   556→                          </div>
   557→                          <h3 className="text-sm font-bold text-slate-800">Vehicle</h3>
   558→                        </div>
   559→                      </div>
   560→                      <div className="p-4">
   561→                        <div className="flex items-start gap-4">
   562→                          {deal.vehicle.primaryImageUrl && (
   563→                            <img
   564→                              src={deal.vehicle.primaryImageUrl}
   565→                              alt={`${deal.vehicle.make} ${deal.vehicle.model}`}
   566→                              className="w-20 h-20 object-cover rounded-xl"
   567→                            />
   568→                          )}
