     1→import { useState, useEffect, useCallback } from "react";
     2→import { toast } from "react-hot-toast";
     3→import ContactPicker from "@/components/ContactPicker";
     4→
     5→// Format currency
     6→const formatCurrency = (amount) => {
     7→  if (amount === null || amount === undefined) return "—";
     8→  return new Intl.NumberFormat("en-GB", {
     9→    style: "currency",
    10→    currency: "GBP",
    11→  }).format(amount);
    12→};
    13→
    14→// Format date helper
    15→const formatDate = (date) => {
    16→  if (!date) return "";
    17→  return new Date(date).toLocaleDateString("en-GB", {
    18→    day: "numeric",
    19→    month: "short",
    20→    year: "numeric",
    21→  });
    22→};
    23→
    24→// Format datetime helper
    25→const formatDateTime = (date) => {
    26→  if (!date) return "";
    27→  return new Date(date).toLocaleString("en-GB", {
    28→    day: "numeric",
    29→    month: "short",
    30→    year: "numeric",
    31→    hour: "2-digit",
    32→    minute: "2-digit",
    33→  });
    34→};
    35→
    36→// Status labels and colors
    37→const STATUS_CONFIG = {
    38→  DRAFT: { label: "Draft", color: "bg-slate-100 text-slate-600" },
    39→  DEPOSIT_TAKEN: { label: "Deposit Taken", color: "bg-amber-100 text-amber-700" },
    40→  INVOICED: { label: "Invoiced", color: "bg-blue-100 text-blue-700" },
    41→  DELIVERED: { label: "Delivered", color: "bg-purple-100 text-purple-700" },
    42→  COMPLETED: { label: "Completed", color: "bg-emerald-100 text-emerald-700" },
    43→  CANCELLED: { label: "Cancelled", color: "bg-red-100 text-red-700" },
    44→};
    45→
    46→const VAT_SCHEME_LABELS = {
    47→  MARGIN: "Margin Scheme",
    48→  VAT_QUALIFYING: "VAT Qualifying",
    49→  NO_VAT: "No VAT",
    50→};
    51→
    52→/**
    53→ * DealDrawer - Display and manage deal details
    54→ *
    55→ * Props:
    56→ * - dealId: The ID of the deal to display
    57→ * - isOpen: Whether the drawer is open
    58→ * - onClose: Function to call when closing the drawer
    59→ * - onUpdate: Function to call when deal is updated
    60→ */
    61→export default function DealDrawer({
    62→  dealId,
    63→  isOpen,
    64→  onClose,
    65→  onUpdate,
    66→}) {
    67→  const [deal, setDeal] = useState(null);
    68→  const [isLoading, setIsLoading] = useState(false);
    69→  const [fetchError, setFetchError] = useState(null);
    70→  const [activeTab, setActiveTab] = useState("overview");
    71→  const [actionLoading, setActionLoading] = useState(null);
    72→
    73→  // DEBUG: Log props changes
    74→  useEffect(() => {
    75→    console.log("[DealDrawer] Props:", { dealId, isOpen });
    76→  }, [dealId, isOpen]);
    77→
    78→  // DEBUG: Log mount/unmount
    79→  useEffect(() => {
    80→    console.log("[DealDrawer] MOUNTED");
    81→    return () => console.log("[DealDrawer] UNMOUNTED");
    82→  }, []);
    83→
    84→  // Deposit modal state
    85→  const [showDepositModal, setShowDepositModal] = useState(false);
    86→  const [depositAmount, setDepositAmount] = useState("");
    87→  const [depositMethod, setDepositMethod] = useState("CARD");
    88→  const [depositReference, setDepositReference] = useState("");
    89→
    90→  // Customer editing state
    91→  const [isEditingCustomer, setIsEditingCustomer] = useState(false);
    92→
    93→  // Add-on state
    94→  const [showAddOnPicker, setShowAddOnPicker] = useState(false);
    95→  const [availableAddOns, setAvailableAddOns] = useState([]);
    96→  const [addOnsLoading, setAddOnsLoading] = useState(false);
    97→  const [addOnQty, setAddOnQty] = useState(1);
    98→  const [addOnPrice, setAddOnPrice] = useState("");
    99→
   100→  // Fetch available add-on products
   101→  const fetchAddOnProducts = async () => {
   102→    setAddOnsLoading(true);
   103→    try {
   104→      const res = await fetch("/api/addons");
   105→      if (res.ok) {
   106→        const data = await res.json();
   107→        setAvailableAddOns(data);
   108→      }
   109→    } catch (error) {
   110→      console.error("Failed to fetch add-ons:", error);
   111→    } finally {
   112→      setAddOnsLoading(false);
   113→    }
   114→  };
   115→
   116→  // Add an add-on to the deal
   117→  const handleAddAddOn = async (addOnProduct) => {
   118→    if (!deal) return;
   119→
   120→    const price = addOnPrice ? parseFloat(addOnPrice) : addOnProduct.defaultPriceNet;
   121→
   122→    const newAddOn = {
   123→      addOnProductId: addOnProduct.id,
   124→      name: addOnProduct.name,
   125→      category: addOnProduct.category,
   126→      qty: addOnQty,
   127→      unitPriceNet: price,
   128→      vatTreatment: addOnProduct.vatTreatment || "STANDARD",
   129→      vatRate: addOnProduct.vatRate || 0.2,
   130→    };
   131→
   132→    setActionLoading("addon");
   133→    try {
   134→      const updatedAddOns = [...(deal.addOns || []), newAddOn];
   135→      const res = await fetch(`/api/deals/${dealId}`, {
   136→        method: "PUT",
   137→        headers: { "Content-Type": "application/json" },
   138→        body: JSON.stringify({ addOns: updatedAddOns }),
   139→      });
   140→
   141→      if (!res.ok) {
   142→        const err = await res.json();
   143→        throw new Error(err.error || "Failed to add add-on");
   144→      }
   145→
   146→      toast.success("Add-on added");
   147→      setShowAddOnPicker(false);
   148→      setAddOnQty(1);
   149→      setAddOnPrice("");
   150→      fetchDeal();
   151→      onUpdate?.();
   152→    } catch (error) {
   153→      toast.error(error.message);
   154→    } finally {
   155→      setActionLoading(null);
   156→    }
   157→  };
   158→
   159→  // Remove an add-on from the deal
   160→  const handleRemoveAddOn = async (index) => {
   161→    if (!deal) return;
   162→
   163→    setActionLoading("addon");
   164→    try {
   165→      const updatedAddOns = deal.addOns.filter((_, i) => i !== index);
   166→      const res = await fetch(`/api/deals/${dealId}`, {
   167→        method: "PUT",
   168→        headers: { "Content-Type": "application/json" },
   169→        body: JSON.stringify({ addOns: updatedAddOns }),
   170→      });
   171→
   172→      if (!res.ok) {
   173→        const err = await res.json();
   174→        throw new Error(err.error || "Failed to remove add-on");
   175→      }
   176→
   177→      toast.success("Add-on removed");
   178→      fetchDeal();
   179→      onUpdate?.();
   180→    } catch (error) {
   181→      toast.error(error.message);
   182→    } finally {
   183→      setActionLoading(null);
   184→    }
   185→  };
   186→
   187→  // Lock body scroll when drawer is open
   188→  useEffect(() => {
   189→    if (isOpen) {
   190→      const scrollY = window.scrollY;
   191→      const html = document.documentElement;
   192→
   193→      document.body.style.position = "fixed";
   194→      document.body.style.top = `-${scrollY}px`;
   195→      document.body.style.left = "0";
   196→      document.body.style.right = "0";
   197→      document.body.style.width = "100%";
   198→      document.body.style.overflow = "hidden";
   199→      document.body.style.touchAction = "none";
   200→
   201→      html.style.overflow = "hidden";
   202→
   203→      return () => {
   204→        document.body.style.position = "";
   205→        document.body.style.top = "";
   206→        document.body.style.left = "";
   207→        document.body.style.right = "";
   208→        document.body.style.width = "";
   209→        document.body.style.overflow = "";
   210→        document.body.style.touchAction = "";
   211→
   212→        html.style.overflow = "";
   213→
   214→        window.scrollTo(0, scrollY);
   215→      };
   216→    }
   217→  }, [isOpen]);
   218→
   219→  useEffect(() => {
   220→    if (isOpen && dealId) {
   221→      fetchDeal();
   222→    } else if (!dealId) {
   223→      // Reset state when dealId is cleared
   224→      setDeal(null);
   225→      setFetchError(null);
   226→    }
   227→  }, [isOpen, dealId]);
   228→
   229→  const fetchDeal = async () => {
   230→    setIsLoading(true);
   231→    setFetchError(null);
   232→    try {
   233→      const res = await fetch(`/api/deals/${dealId}`);
   234→      if (!res.ok) {
   235→        const errData = await res.json().catch(() => ({}));
   236→        throw new Error(errData.error || `Failed to load deal (${res.status})`);
   237→      }
   238→      const data = await res.json();
   239→      setDeal(data);
   240→    } catch (error) {
   241→      console.error("[DealDrawer] Fetch error:", error);
   242→      setFetchError(error.message || "Failed to load deal details");
   243→      // Don't show toast here - we'll show error in UI instead
   244→    } finally {
   245→      setIsLoading(false);
   246→    }
   247→  };
   248→
   249→  // Calculate totals
   250→  const calculateTotals = () => {
   251→    if (!deal) return {};
   252→
   253→    const addOnsNetTotal = (deal.addOns || []).reduce(
   254→      (sum, a) => sum + (a.unitPriceNet * (a.qty || 1)),
   255→      0
   256→    );
   257→    const addOnsVatTotal = (deal.addOns || []).reduce((sum, a) => {
   258→      if (a.vatTreatment === "STANDARD") {
   259→        return sum + a.unitPriceNet * (a.qty || 1) * (a.vatRate || 0.2);
   260→      }
   261→      return sum;
   262→    }, 0);
   263→
   264→    let subtotal, totalVat, grandTotal;
   265→
   266→    if (deal.vatScheme === "VAT_QUALIFYING") {
   267→      subtotal = (deal.vehiclePriceNet || 0) + addOnsNetTotal;
   268→      totalVat = (deal.vehicleVatAmount || 0) + addOnsVatTotal;
   269→      grandTotal = subtotal + totalVat;
   270→    } else {
   271→      subtotal = (deal.vehiclePriceGross || 0) + addOnsNetTotal + addOnsVatTotal;
   272→      totalVat = 0;
   273→      grandTotal = subtotal;
   274→    }
   275→
   276→    const totalPaid = (deal.payments || [])
   277→      .filter((p) => !p.isRefunded)
   278→      .reduce((sum, p) => sum + p.amount, 0);
   279→
   280→    const pxNetValue = deal.partExchange
   281→      ? (deal.partExchange.allowance || 0) - (deal.partExchange.settlement || 0)
   282→      : 0;
   283→
   284→    const balanceDue = grandTotal - totalPaid - pxNetValue;
   285→
   286→    return {
   287→      addOnsNetTotal,
   288→      addOnsVatTotal,
   289→      subtotal,
   290→      totalVat,
   291→      grandTotal,
   292→      totalPaid,
   293→      pxNetValue,
   294→      balanceDue,
   295→    };
   296→  };
   297→
   298→  // Deal actions
   299→  const handleTakeDeposit = async () => {
   300→    if (!depositAmount || parseFloat(depositAmount) <= 0) {
   301→      toast.error("Please enter a valid deposit amount");
   302→      return;
   303→    }
   304→
   305→    setActionLoading("deposit");
   306→    try {
   307→      const res = await fetch(`/api/deals/${dealId}/take-deposit`, {
   308→        method: "POST",
   309→        headers: { "Content-Type": "application/json" },
   310→        body: JSON.stringify({
   311→          amount: parseFloat(depositAmount),
   312→          method: depositMethod,
   313→          reference: depositReference,
   314→        }),
   315→      });
   316→
   317→      if (!res.ok) {
   318→        const err = await res.json();
   319→        throw new Error(err.error || "Failed to record deposit");
   320→      }
   321→
   322→      const result = await res.json();
   323→      toast.success(`Deposit recorded: ${result.documentNumber}`);
   324→      setShowDepositModal(false);
   325→      setDepositAmount("");
   326→      setDepositReference("");
   327→      fetchDeal();
   328→      onUpdate?.();
   329→    } catch (error) {
   330→      toast.error(error.message);
   331→    } finally {
   332→      setActionLoading(null);
   333→    }
   334→  };
   335→
   336→  const handleGenerateInvoice = async () => {
   337→    setActionLoading("invoice");
   338→    try {
   339→      const res = await fetch(`/api/deals/${dealId}/generate-invoice`, {
   340→        method: "POST",
   341→        headers: { "Content-Type": "application/json" },
   342→      });
   343→
   344→      if (!res.ok) {
   345→        const err = await res.json();
   346→        throw new Error(err.error || "Failed to generate invoice");
   347→      }
   348→
   349→      const result = await res.json();
   350→      toast.success(`Invoice generated: ${result.documentNumber}`);
   351→      fetchDeal();
   352→      onUpdate?.();
   353→    } catch (error) {
   354→      toast.error(error.message);
   355→    } finally {
   356→      setActionLoading(null);
   357→    }
   358→  };
   359→
   360→  const handleMarkDelivered = async () => {
   361→    setActionLoading("delivered");
   362→    try {
   363→      const res = await fetch(`/api/deals/${dealId}/mark-delivered`, {
   364→        method: "POST",
   365→        headers: { "Content-Type": "application/json" },
   366→      });
   367→
   368→      if (!res.ok) {
   369→        const err = await res.json();
   370→        throw new Error(err.error || "Failed to mark as delivered");
   371→      }
   372→
   373→      toast.success("Deal marked as delivered");
   374→      fetchDeal();
   375→      onUpdate?.();
   376→    } catch (error) {
   377→      toast.error(error.message);
   378→    } finally {
   379→      setActionLoading(null);
   380→    }
   381→  };
   382→
   383→  const handleMarkCompleted = async () => {
   384→    setActionLoading("completed");
   385→    try {
   386→      const res = await fetch(`/api/deals/${dealId}/mark-completed`, {
   387→        method: "POST",
   388→        headers: { "Content-Type": "application/json" },
   389→      });
   390→
   391→      if (!res.ok) {
   392→        const err = await res.json();
   393→        throw new Error(err.error || "Failed to complete deal");
   394→      }
   395→
   396→      toast.success("Deal completed");
   397→      fetchDeal();
   398→      onUpdate?.();
   399→    } catch (error) {
   400→      toast.error(error.message);
   401→    } finally {
   402→      setActionLoading(null);
   403→    }
   404→  };
   405→
   406→  const handleCancelDeal = async () => {
   407→    if (!confirm("Are you sure you want to cancel this deal?")) return;
   408→
   409→    setActionLoading("cancel");
   410→    try {
   411→      const res = await fetch(`/api/deals/${dealId}`, {
   412→        method: "DELETE",
   413→        headers: { "Content-Type": "application/json" },
   414→        body: JSON.stringify({ cancelReason: "Cancelled by user" }),
   415→      });
   416→
   417→      if (!res.ok) {
   418→        const err = await res.json();
   419→        throw new Error(err.error || "Failed to cancel deal");
   420→      }
   421→
   422→      toast.success("Deal cancelled");
   423→      fetchDeal();
   424→      onUpdate?.();
   425→    } catch (error) {
   426→      toast.error(error.message);
   427→    } finally {
   428→      setActionLoading(null);
   429→    }
   430→  };
   431→
   432→  const handleUpdateCustomer = async (contactId) => {
   433→    if (!contactId) return;
   434→
   435→    setActionLoading("customer");
   436→    try {
   437→      const res = await fetch(`/api/deals/${dealId}`, {
   438→        method: "PUT",
   439→        headers: { "Content-Type": "application/json" },
   440→        body: JSON.stringify({ soldToContactId: contactId }),
   441→      });
   442→
   443→      if (!res.ok) {
   444→        const err = await res.json();
   445→        throw new Error(err.error || "Failed to update customer");
   446→      }
   447→
   448→      toast.success("Customer updated");
   449→      setIsEditingCustomer(false);
   450→      fetchDeal();
   451→      onUpdate?.();
   452→    } catch (error) {
   453→      toast.error(error.message);
   454→    } finally {
   455→      setActionLoading(null);
   456→    }
   457→  };
   458→
   459→  if (!isOpen) return null;
   460→
   461→  const totals = calculateTotals();
   462→  const statusConfig = STATUS_CONFIG[deal?.status] || STATUS_CONFIG.DRAFT;
   463→
   464→  return (
   465→    <div
   466→      className="fixed inset-0 flex justify-end z-50 drawer-locked"
   467→      style={{ touchAction: "none", overscrollBehavior: "none" }}
   468→    >
   469→      {/* Backdrop */}
   470→      <div
   471→        className="absolute inset-0 bg-black/40 backdrop-blur-sm transition-opacity duration-300"
   472→        onClick={onClose}
   473→      />
   474→
   475→      {/* Drawer */}
   476→      <div
   477→        className="relative bg-[#f8fafc] flex flex-col w-full max-w-3xl min-w-0 translate-x-0 overflow-x-hidden shadow-2xl"
   478→        style={{
   479→          height: "100dvh",
   480→          maxHeight: "100dvh",
   481→          touchAction: "pan-y",
   482→        }}
   483→      >
   484→        {isLoading ? (
   485→          <div className="flex items-center justify-center h-full">
   486→            <span className="loading loading-spinner loading-lg text-[#0066CC]"></span>
   487→          </div>
   488→        ) : fetchError ? (
   489→          <div className="flex flex-col items-center justify-center h-full p-6">
   490→            <div className="w-16 h-16 bg-red-100 rounded-full flex items-center justify-center mb-4">
   491→              <svg className="w-8 h-8 text-red-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   492→                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z" />
   493→              </svg>
   494→            </div>
   495→            <p className="text-lg font-semibold text-slate-900 mb-1">Failed to load deal</p>
   496→            <p className="text-sm text-slate-500 text-center mb-4">{fetchError}</p>
   497→            <div className="flex gap-2">
   498→              <button className="btn btn-ghost" onClick={onClose}>
   499→                Close
   500→              </button>
   501→              <button className="btn bg-[#0066CC] hover:bg-[#0052a3] text-white border-none" onClick={fetchDeal}>
   502→                <svg className="w-4 h-4 mr-1" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   503→                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M4 4v5h.582m15.356 2A8.001 8.001 0 004.582 9m0 0H9m11 11v-5h-.581m0 0a8.003 8.003 0 01-15.357-2m15.357 2H15" />
   504→                </svg>
   505→                Retry
   506→              </button>
   507→            </div>
   508→          </div>
   509→        ) : !deal ? (
   510→          <div className="flex flex-col items-center justify-center h-full p-6">
   511→            <div className="w-16 h-16 bg-slate-100 rounded-full flex items-center justify-center mb-4">
   512→              <svg className="w-8 h-8 text-slate-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   513→                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 12h6m-6 4h6m2 5H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z" />
   514→              </svg>
   515→            </div>
   516→            <p className="text-lg font-semibold text-slate-900 mb-1">Deal not found</p>
   517→            <p className="text-sm text-slate-500 text-center mb-4">This deal may have been deleted or you may not have access</p>
   518→            <button className="btn btn-ghost" onClick={onClose}>
   519→              Close
   520→            </button>
   521→          </div>
   522→        ) : (
   523→          <>
   524→            {/* Header */}
   525→            <div className="sticky top-0 bg-white border-b border-slate-100 px-4 md:px-6 py-4 md:py-5 z-10 shrink-0 shadow-sm">
   526→              <div className="flex items-center justify-between">
   527→                <div className="flex items-center gap-3">
   528→                  <button
   529→                    className="flex items-center justify-center w-9 h-9 rounded-xl bg-slate-100 hover:bg-[#0066CC]/10 text-slate-500 hover:text-[#0066CC] transition-all"
   530→                    onClick={onClose}
   531→                  >
   532→                    <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   533→                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M15 19l-7-7 7-7" />
   534→                    </svg>
   535→                  </button>
   536→                  <div>
   537→                    <h2 className="text-lg md:text-xl font-bold text-slate-900">
   538→                      Deal #{deal.dealNumber || "—"}
   539→                    </h2>
   540→                    <div className="flex items-center gap-2 mt-1">
   541→                      <span className={`inline-flex items-center px-2.5 py-1 rounded-lg text-xs font-semibold ${statusConfig.color}`}>
   542→                        {statusConfig.label}
   543→                      </span>
   544→                      {deal.vehicle && (
   545→                        <span className="inline-flex items-center px-2.5 py-1 rounded-lg bg-slate-900 text-white text-xs font-mono font-bold tracking-wider">
   546→                          {deal.vehicle.regCurrent}
   547→                        </span>
   548→                      )}
   549→                    </div>
   550→                  </div>
   551→                </div>
   552→                <button
   553→                  className="flex items-center justify-center w-9 h-9 rounded-xl bg-slate-100 hover:bg-red-50 text-slate-400 hover:text-red-500 transition-all"
   554→                  onClick={onClose}
   555→                >
   556→                  <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
   557→                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
   558→                  </svg>
   559→                </button>
   560→              </div>
   561→            </div>
   562→
   563→            {/* Tabs */}
   564→            <div className="sticky top-[73px] md:top-[89px] bg-white border-b border-slate-100 px-4 py-3 z-10 shrink-0">
   565→              {/* Mobile: Dropdown selector */}
   566→              <div className="md:hidden">
   567→                <select
