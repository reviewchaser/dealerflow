     1→/**
     2→ * Portal Component - Renders children into document.body
     3→ * This escapes the DOM hierarchy to avoid CSS stacking context issues
     4→ * (transform, filter, will-change on ancestors break fixed positioning)
     5→ */
     6→
     7→import { useEffect, useState } from "react";
     8→import { createPortal } from "react-dom";
     9→
    10→export function Portal({ children }) {
    11→  const [mounted, setMounted] = useState(false);
    12→
    13→  useEffect(() => {
    14→    setMounted(true);
    15→    return () => setMounted(false);
    16→  }, []);
    17→
    18→  if (!mounted) return null;
    19→
    20→  return createPortal(children, document.body);
    21→}
    22→
    23→export default Portal;
