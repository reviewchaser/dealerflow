   195→        quickAddButtonRef.current &&

   289→            localStorage.setItem(LOGO_CACHE_KEY, data.logoUrl);
   290→          } else {
   291→            localStorage.removeItem(LOGO_CACHE_KEY);
   292→          }
   293→          if (data.name) {
   294→            localStorage.setItem(DEALER_NAME_CACHE_KEY, data.name);
   295→          }
   296→        } catch {
   297→          // localStorage not available
   298→        }
   299→
   300→        // Redirect to onboarding wizard if not completed (skip if already on onboarding page)
   301→        if (data && !data.completedOnboarding && !router.pathname.startsWith("/onboarding")) {
   302→          router.push(getPath("/onboarding"));
   303→        }
   304→      })
   305→      .catch(() => {

   307→      });
